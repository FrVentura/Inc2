void mirror (ABin *)

void mirror (ABin *a){
    ABin aa = *a;
    if (aa==NULL)
        return;
    mirror(&(aa->esq));
    mirror(&(aa->dir));
    ABin temp = aa->esq;
    aa->esq = aa->dir;
    aa->dir = temp;
}

int depth (ABin, int)

int depth (ABin a, int x){
    int level = 1;
    int tester = 0;
    if (a==NULL)
        return -1;
    if (x > a->valor){
        tester = depth(a->dir,x);
        if (tester > -1)
            level = 1 + depth(a->dir, x);
        else
            level =-1;
    }
    else if (x < a->valor){
        tester = depth(a->esq,x);
        if (tester > -1)
            level = 1 + depth(a->esq, x);
        else
            level =-1;
    }
    return level;
}

int pruneAB (ABin *, int)

int pruneAB (ABin *a, int l){
    int level = l;
    if (level==0){
        return freeAB(*a);
        (*a)->esq = NULL;
        (*a)->dir = NULL;
    }
    level --;

    if ((*a)->esq && (*a)->dir)
        return  (pruneAB(&((*a)->esq), level ) + pruneAB(&((*a)->dir), level ));
    else if (!(*a)->esq && (*a)->dir)
        return pruneAB(&((*a)->dir), level);
    else if ((*a)->esq && !(*a)->dir)
        return pruneAB(&((*a)->esq), level);

    //if (!(*a)->esq && !(*a)->dir)
    return 0;
}

int min(int a, int b){
    if (a<b)
        return a;
    return b;
}

int iguais (ABin a, ABin b){
    if ((!a && b) || (a && !b) )
        return 0;
    else if (a && b && a->valor != b->valor)
        return 0;
    int r = 1;
    if (a && b)
        r = min(iguais(a->esq,b->esq), iguais(a->dir,b->dir));
    return r;
}

LInt nivel (ABin, int)

LInt nivel (ABin a, int n){
    LInt l;
    if (!a)
        return NULL;
    if (n==1){
        l = newLInt(a->valor, NULL);
        return l;
    }
    LInt head;
    l = nivel(a->esq, n-1);
    if (l){
        head = l;
        while (l->prox)
            l = l->prox;
        l->prox = nivel(a->dir, n-1);
    }
    else {
        l = nivel(a->dir, n-1);
        head = l;
    }
    return head;
}

nivelArray (ABin, int, int v[])

int nivelArray (ABin a, int n, int v[]){
    int* temp = v;
    if (a==NULL)
        return 0;
    if (n==1){
        v[0] = a->valor;
        return 1;
    }
    else{
        n--;
        v+=nivelArray(a->esq,n,v);
        v+=nivelArray(a->dir,n,v);
    }
    return v - temp;
}

int dumpABin (ABin, int v[], int)

int dumpAbin (ABin a, int v[], int N){
    int temp1, temp2;
    if (a==NULL|| N<0)
        return 0;

    temp1=dumpAbin(a->esq, v, N);
    N-=temp1;
    v+=temp1;
    *v = (a->valor);
    N--;
    temp2=dumpAbin(a->dir,v+1,N);

    return temp2+temp1+1;
}

ABin somasAB (ABin a){
    if (a==NULL)
        return NULL;
    ABin r = newABin(a->valor,NULL,NULL);

    int z = a->valor;

    if (a->esq && a->dir)
        z= somasAB(a->esq)->valor + somasAB(a->dir)->valor;
    else if (a->esq == NULL && a->dir)
        z = somasAB(a->dir)->valor;
    else if (a->esq && a->dir == NULL)
        z = somasAB(a->esq)->valor;
    else if (a->esq == NULL && a->dir==NULL)
        z = a->valor;

    r = newABin(z,somasAB(a->esq),somasAB(a->dir));

    return r;
}

int add (ABin *, int)

int add (ABin *a, int x){
    if (!*a){
        *a = newABin (x,NULL,NULL);
        return 0;
    }
    else if ((*a)->valor == x){
        return 1;
    }
    else if ((*a)->valor < x){
        add (&((*a)->dir),x);
    }
    else if ((*a)->valor > x){
        add (&((*a)->esq),x);
    }
    return 1;
}

void removeMaiorAB (ABin *a){
    ABin iter = *a;
    if (!iter)
        return;
    ABin ante = iter;
    iter = iter->dir;
    while (iter->dir){
        ante = iter;
        iter = iter->dir;
    }
    free(iter);
    ante->dir = NULL;
}

int maiores (ABin a, int x){
    if (!a)
        return 0;
    if (a->valor > x)
        return 1 + maiores(a->esq, x) + maiores(a->dir, x);
    else
        return maiores(a->esq, x) + maiores (a->dir, x);
}
